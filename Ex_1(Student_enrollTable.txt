
-- STUDENTS TABLE

CREATE TABLE STUDENTS_KUN (
SID NUMERIC(10),
NAME VARCHAR2(50),
LOGIN VARCHAR2(50),
AGE NUMERIC(10),
GPA NUMERIC(10,2),
CONSTRAINT PK_STUDENTS_KUN PRIMARY KEY(SID));

-- ENROLLED TABLE

CREATE TABLE ENROLLED_KUN(
SID NUMERIC(10),
CID VARCHAR2(50),
GRADE VARCHAR2(10),
CONSTRAINT PK_ENROLLED_KUN PRIMARY KEY(CID),
CONSTRAINT FK_ENROLLED_KUN FOREIGN KEY(SID) REFERENCES STUDENTS_KUN(SID));

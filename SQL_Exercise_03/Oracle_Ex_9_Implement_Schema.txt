-- Table supplier_kun

CREATE TABLE SUPPLIER_KUN(
SUPPLIER_ID VARCHAR2(10),
COMPANY_NAME VARCHAR2(50),
CONTACT_NAME VARCHAR2(50),
CONTACT_TITLE VARCHAR2(50),
ADDRESS VARCHAR2(50),
CITY VARCHAR2(50),
REGION VARCHAR2(50),
CONSTRAINT PK_SUPPLIER_KUN PRIMARY KEY(SUPPLIER_ID));

--Table categories_kun

CREATE TABLE CATEGORIES_KUN(
CATEGORY_ID VARCHAR2(10),
CATEGORY_NAME VARCHAR2(50),
DESCRIPTION VARCHAR2(50),
PICTURE VARCHAR2(50),
CONSTRAINT PK_CATEGORIES_KUN PRIMARY KEY(CATEGORY_ID));

-- Table PRODUCTS_KUN 

CREATE TABLE PRODUCTS_KUN(
PRODUCT_ID VARCHAR2(10),
PRODUCT_NAME VARCHAR2(50),
SUPPLIER_ID VARCHAR2(10),
CATEGORY_ID VARCHAR2(10),
QUANTITY_PER_UNIT NUMERIC(10,2),
UNIT_PRICE NUMERIC(10,2),
UNITS_IN_STOCK NUMERIC(10,2),
UNITS_ON_ORDER VARCHAR2(50),
REORDER_LEVEL VARCHAR2(50),
DISCOUNTINUED VARCHAR2(50),
CONSTRAINT PK_PRODUCTS_KUN PRIMARY KEY(PRODUCT_ID),
CONSTRAINT FK_PRODUCTS_KUN1 FOREIGN KEY(SUPPLIER_ID) REFERENCES SUPPLIER_KUN(SUPPLIER_ID),
CONSTRAINT FK_PRODUCTS_KUN2 FOREIGN KEY(CATEGORY_ID) REFERENCES CATEGORIES_KUN(CATEGORY_ID));

--Table ORDER_DETAILS_KUN 
 
CREATE TABLE ORDER_DETAILS_KUN(
ORDER_DATAILS_ID VARCHAR2(10),
PRODUCT_ID VARCHAR2(10),
UNIT_PRICE NUMERIC(10,2),
QUANTITY NUMERIC(10),
DISCOUNT NUMERIC(10,2),
CONSTRAINT PK_ORDER_DETAILS_KUN PRIMARY KEY(ORDER_ID),
CONSTRAINT FK_ORDER_DETAILS_KUN FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS_KUN(PRODUCT_ID));

-- Table EMPLOYEESS_KUN

CREATE TABLE EMPLOYEESS_KUN(
EMPLOYEE_ID VARCHAR2(10),
LAST_NAME VARCHAR2(50),
FIST_NAME VARCHAR2(50),
TITLE VARCHAR2(50),
TITLE_OF_COURTER VARCHAR2(50),
BIRTH_DATE DATE,
HIRE_DATE DATE,
ADDRESS VARCHAR2(50),
CITY VARCHAR2(50),
CONSTRAINT PK_EMPLOYEESS_KUN PRIMARY KEY(EMPLOYEE_ID));

-- Table CUSTOMERS_KUN

CREATE TABLE CUSTOMERS_KUN(
CUSTOMER_ID VARCHAR2(10),
COMPANY_NAME VARCHAR2(50),
CONTACT_NAME VARCHAR2(50),
CONTACT_TITLE VARCHAR2(50),
ADDRESS VARCHAR2(50),
CITY VARCHAR2(50),
REGION VARCHAR2(50),
CONSTRAINT PK_CUSTOMERS_KUN PRIMARY KEY(CUSTOMER_ID)
);


-- Table SHIPPERS_KUN

CREATE TABLE SHIPPERS_KUN(
SHIPPER_ID VARCHAR2(10),
COMPANY_NAME VARCHAR2(50),
PHONE VARCHAR2(50),
CONSTRAINT PK_SHIPPERS_KUN PRIMARY KEY(SHIPPER_ID)
);

-- Table ORDERS_KUN

CREATE TABLE ORDERS_KUN(
ORDERS_ID VARCHAR(10),
ORDER_DATAILS_ID VARCHAR2(10),
CUSTOMER_ID VARCHAR2(10),
EMPLOYEE_ID VARCHAR2(10),
ORDER_DATE DATE,
REQUIRE_DATE DATE,
SHIPPED_DATE DATE,
SHIPVIA VARCHAR2(50),
FREIGHT VARCHAR2(50),
SHIIP_NAME VARCHAR2(50),
SHIP_ADDRESS VARCHAR2(50),
SHIP_CITY VARCHAR2(50),
SHIP_REGION VARCHAR2(50),
SHIP_POSTAL_CODE NUMERIC(10),
CONSTRAINT PK_ORDERS_KUN PRIMARY KEY(ORDERS_ID),
CONSTRAINT FK_ORDERS_ID1 FOREIGN KEY(ORDER_DATAILS_ID) REFERENCES ORDER_DETAILS_KUN(ORDER_DATAILS_ID),
CONSTRAINT FK_ORDERS_ID2 FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMERS_KUN(CUSTOMER_ID),
CONSTRAINT FK_ORDERS_ID3 FOREIGN KEY(EMPLOYEE_ID) REFERENCES EMPLOYEESS_KUN(EMPLOYEE_ID),
CONSTRAINT FK_ORDERS_ID4 FOREIGN KEY(SHIPVIA) REFERENCES SHIPPERS_KUN(SHIPPER_ID)
);
